read_olympedia <- function(url, ...){
  if(stringr::str_starts(url, "/")) {
    url <- stringr::str_remove(url, "^/")
  }

  cache_dir <- fs::path("data", "mp04", "olympedia_cache")
  if(!fs::dir_exists(cache_dir)) fs::dir_create(cache_dir, recurse = TRUE)

  cache_path <- fs::path(
    cache_dir,
    url |> as.character() |> stringr::str_replace_all("/", "_")
  )

  if(!fs::file_exists(cache_path)){
    src_url <- paste0("https://www.olympedia.org/", url)
    slow_download(src_url, cache_path, mode = "wb")
  }

  rvest::read_html(cache_path)
}

read_editions <- function() {
  pg <- read_olympedia("editions")

  tables <- pg |> html_elements("table")

  parse_editions_table <- function(tbl, season_label) {
    rows <- tbl |> html_elements("tr")

    map_dfr(rows, function(row) {
      tds <- row |> html_elements("td")
      if(length(tds) < 8) return(NULL)

      edition_link <- row |> html_element("a[href^='/editions/']")
      edition_url <- if(length(edition_link) == 0) NA_character_ else html_attr(edition_link, "href")
      edition_id  <- str_extract(edition_url, "\\d+$") |> as.integer()

      olympiad_roman <- tds[[1]] |> html_text2() |> str_trim()
      year_txt <- tds[[2]] |> html_text2() |> str_trim()
      city <- tds[[3]] |> html_text2() |> str_trim()
      note <- tds[[8]] |> html_text2() |> str_squish()

      year <- suppressWarnings(as.integer(year_txt))

      flag_img <- tds[[4]] |> html_element("img")
      flag_src <- if(length(flag_img) == 0) NA_character_ else html_attr(flag_img, "src")
      host_country_code <- str_extract(flag_src, "(?<=/)[A-Z]{3}(?=\\.png$)")

      if(is.na(edition_id) || is.na(year) || olympiad_roman == "") return(NULL)

      tibble(
        season = season_label,
        year = year,
        city = city,
        olympiad_roman = olympiad_roman,
        olympiad_id = roman_to_integer(olympiad_roman),
        edition_id = edition_id,
        edition_url = edition_url,
        host_country_code = host_country_code,
        note = note
      )
    })
  }

  bind_rows(
    parse_editions_table(tables[[1]], "Summer"),
    parse_editions_table(tables[[2]], "Winter")
  ) |>
    distinct(edition_id, .keep_all = TRUE) |>
    filter(!str_detect(str_to_lower(note), "not held")) |>
    select(-note) |>
    arrange(year, season)
}

read_edition_disciplines <- function(edition_url) {
  pg <- read_olympedia(edition_url)

  sport_table <- pg |>
    html_element(xpath = "//h2[normalize-space()='Medal Disciplines']/following-sibling::table[1]")

  if(length(sport_table) == 0) {
    return(tibble(
      discipline = character(),
      discipline_url = character(),
      discipline_code = character()
    ))
  }

  links <- sport_table |> html_elements("a")

  tibble(
    discipline = html_text2(links) |> str_squish(),
    discipline_url = html_attr(links, "href")
  ) |>
    filter(str_detect(discipline_url, "/sports/")) |>
    mutate(
      discipline_code = str_extract(discipline_url, "(?<=/sports/)[A-Z0-9]+")
    ) |>
    distinct()
}

read_discipline_medal_table <- function(discipline_url) {
  pg <- read_olympedia(discipline_url)

  medal_table_node <- pg |>
    html_element(xpath = "//h2[normalize-space()='Medal table']/following-sibling::table[1]")

  if(length(medal_table_node) == 0) {
    return(tibble(
      competitor = character(),
      country_code = character(),
      gold = integer(),
      silver = integer(),
      bronze = integer(),
      total = integer()
    ))
  }

  medal_tbl <- medal_table_node |> html_table(fill = TRUE)

  # At least 6 columns: competitor, 3‑letter code, then Gold, Silver, Bronze, Total
  if(ncol(medal_tbl) < 6) {
    return(tibble(
      competitor = character(),
      country_code = character(),
      gold = integer(),
      silver = integer(),
      bronze = integer(),
      total = integer()
    ))
  }

  tibble(
    competitor = as.character(medal_tbl[[1]]),
    country_code = as.character(medal_tbl[[2]]),
    gold = suppressWarnings(as.integer(medal_tbl[[3]])),
    silver = suppressWarnings(as.integer(medal_tbl[[4]])),
    bronze = suppressWarnings(as.integer(medal_tbl[[5]])),
    total = suppressWarnings(as.integer(medal_tbl[[6]]))
  ) |>
    filter(!is.na(country_code), country_code != "")
}

final_medals_df <- olympedia_medals %>%
  # Filter to match specific nations
  filter(country_code %in% c("AUS", "CHN", "FRA", "GBR", "USA")) %>%
  
  # IMPUTATION STEP: Fill in the missing host country based on the city
  mutate(
    actual_host = case_when(
      city %in% c("Melbourne", "Sydney") ~ "AUS",
      city %in% c("Beijing") ~ "CHN",
      city %in% c("Paris", "Chamonix", "Grenoble", "Albertville") ~ "FRA",
      city %in% c("London") ~ "GBR",
      city %in% c("St. Louis", "Los Angeles", "Atlanta", "Squaw Valley", "Lake Placid", "Salt Lake City") ~ "USA",
      .default = "OTHER" # Any city not listed above was hosted by someone else
    ),
    
    # Now our logical test will work!
    olympic_status = ifelse(country_code == actual_host, "Host", "Away")
  ) %>%
  
  # Aggregate total golds per country per year
  group_by(year, country_code, olympic_status) %>%
  summarize(total_golds = sum(gold, na.rm = TRUE), .groups = "drop") %>%
  
  # Convert to a strict factor for plotting
  mutate(olympic_status_factor = factor(olympic_status, levels = c("Host", "Away")))

# Set seed for exact simulation reproducibility
set.seed(9750)
n_sims <- 1e7 # 10 Million simulations for high-precision stability

# Create an empty list to store our final intervals
mc_results <- list()

# Loop through each medal type to simulate their unique distribution properties
for(m in medal_types) {
  
  # 1. Extract and map parameters for: US Baseline
  us_row <- local_summary %>% filter(Medal == m)
  us_mean <- us_row$US_Mean
  us_sd   <- (us_row$US_Upper - us_mean) / qnorm(1 - 0.05/2)
  
  # 2. Extract and map parameters for: Host Raw Increase
  host_mean <- us_row$Host_Inc_Mean
  host_sd   <- (us_row$Host_Upper - host_mean) / qnorm(1 - 0.05/2)
  
  # 3. Extract and map parameters for: New Sport Win Probability
  prop_mean <- us_row$Prop_Mean
  prop_sd   <- (us_row$Prop_Upper - prop_mean) / qnorm(1 - 0.05/2)
  
  # 4. Generate the Monte Carlo normal random distributions
  sim_us_baseline <- rnorm(n_sims, mean = us_mean, sd = us_sd)
  sim_host_boost  <- rnorm(n_sims, mean = host_mean, sd = host_sd)
  sim_new_sport   <- rnorm(n_sims, mean = prop_mean, sd = prop_sd)
  
  # Enforce logical boundary: probability cannot be less than 0 or greater than 1
  sim_new_sport   <- pmax(0, pmin(1, sim_new_sport))
  
  # 5. Run the simulated trials through the official Task 5 formula
  # Formula accounts for the 22 expansion events scheduled for LA 2028
  simulated_2028_medals <- sim_us_baseline + sim_host_boost + (22 * sim_new_sport)
  
  # Enforce logical boundary: a country cannot win a negative number of medals
  simulated_2028_medals <- pmax(0, simulated_2028_medals)
  
  # 6. Extract the precise 95% boundaries (Percentile Method)
  ci_bounds <- quantile(simulated_2028_medals, c(0.025, 0.500, 0.975))
  
  # Save the simulation snapshot
  mc_results[[m]] <- tibble(
    Medal = m,
    Lower_CI = ci_bounds[1],
    Median_Point = ci_bounds[2],
    Upper_CI = ci_bounds[3]
