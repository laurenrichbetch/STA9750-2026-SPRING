# Helper function to clean Census numbers
clean_census_number <- function(x) {
  parse_number(as.character(x), na = c("N", "(X)", "Z", "NA", ""))
}

state_to_state_migration_2024 <- function() {
  dir_path <- "data/mp03"
  if (!dir.exists(dir_path)) dir.create(dir_path, recursive = TRUE)
  file_path <- file.path(dir_path, "migration_2024.xlsx")
  
  if (!file.exists(file_path)) {
    url <- "https://www2.census.gov/programs-surveys/demo/tables/geographic-mobility/2024/state-to-state-migration/State_to_State_Migration_Table_2024_T13.xlsx"
    download.file(url, file_path, mode = "wb")
  }
  
  # 1. Process 'Table'
  table_data <- read_excel(file_path, sheet = "Table", skip = 5, col_names = FALSE) %>%
    select(state_current = 1, state_1y = 2, population = 3) %>%
    mutate(population = clean_census_number(population)) %>%
    filter(!is.na(population), !is.na(state_current)) %>%
    mutate(year_current = 2024, year_1y = 2023)
  
  # 2. Process 'Supplemental'
  supp_data <- read_excel(file_path, sheet = "Supplemental - Current Res", skip = 5, col_names = FALSE) %>%
    select(state_current = 1, stayed_1 = 2, stayed_2 = 3) %>%
    mutate(
      stayed_1 = clean_census_number(stayed_1),
      stayed_2 = clean_census_number(stayed_2),
      population = coalesce(stayed_1, 0) + coalesce(stayed_2, 0)
    ) %>%
    filter(!is.na(state_current)) %>% 
    mutate(
      state_1y = state_current,
      year_current = 2024, year_1y = 2023
    ) %>%
    select(state_current, state_1y, population, year_current, year_1y)
  
  # 3. USE BIND_ROWS (This is the fix)
  final_data <- bind_rows(table_data, supp_data)
  
  return(final_data)
}

  parse_number(as.character(x), na = c("N", "(X)", "Z", "NA", ""))

state_to_state_migration_2023 <- function() {
  file_path <- "data/mp03/migration_2023.xlsx"
  states_with_dc <- c(state.name[1:8], "District of Columbia", state.name[9:50])
  
  # 1. Stationary Data: Start at row 8 to skip all headers
  stayed_data <- read_excel(file_path, range = "A8:F58", col_names = FALSE) %>%
    select(state_current = 1, stayed_1 = 5, stayed_2 = 6) %>%
    mutate(
      population = clean_census_number_2023(stayed_1) + clean_census_number_2023(stayed_2),
      state_1y = state_current,
      year_current = 2023, year_1y = 2022
    ) %>%
    filter(!is.na(population)) # Drop any junk rows
  
  # 2. Flow Data: Start at row 8
  flow_data <- read_excel(file_path, range = "J8:DG58", 
                          col_names = vctrs::vec_interleave(states_with_dc, rep(NA, 51))) %>%
    select(any_of(states_with_dc)) %>%
    mutate(state_current = states_with_dc) %>%
    pivot_longer(cols = -state_current, names_to = "state_1y", values_to = "population") %>%
    mutate(population = clean_census_number_2023(population)) %>%
    filter(!is.na(population), state_current != state_1y) %>% # Drop headers and self-loops
    mutate(year_current = 2023, year_1y = 2022)
  
  return(bind_rows(stayed_data, flow_data))
}

state_to_state_migration <- function() {
  # 1. Fetch data from your two existing functions
  df_2024 <- state_to_state_migration_2024()

  # 2. Combine and filter to keep only official states
  official_states <- c(state.name, "District of Columbia")
  
  combined_data <- bind_rows(df_2024, df_2023) %>%
    # Keep only records where BOTH origin and destination are in your official list
    filter(state_current %in% official_states,
           state_1y %in% official_states) %>%
    # Select only the columns needed for analysis
    select(state_current, state_1y, population, year_current, year_1y)
  
  return(combined_data)
}

metro_to_metro_migration <- function() {

    data_dir <- "data/mp03"
    if (!dir.exists(data_dir)) dir.create(data_dir, recursive = TRUE)

    api_url <- "https://api.census.gov/data/2020/acs/flows?get=MOVEDIN,MOVEDOUT,FULL1_NAME,FULL2_NAME&for=metropolitan%20statistical%20area/micropolitan%20statistical%20area:*"
    dest_file <- file.path(data_dir, "metro-to-metro-migration.json")
    
    if (!file.exists(dest_file)) {
        message("Fetching Metro Data from Census API...")
        # Ensure you have library(httr2) loaded
        httr2::request(api_url) |>
            httr2::req_perform() |>
            httr2::resp_body_string() |>
            writeLines(dest_file)
    }

    # FIX: Replace read_json with fromJSON
    raw_json <- jsonlite::fromJSON(dest_file, simplifyVector = TRUE)
    
    # Convert matrix to data frame and set headers from the first row
    metro_raw <- as.data.frame(raw_json[-1, ], stringsAsFactors = FALSE)
    colnames(metro_raw) <- raw_json[1, ]
    
    # Tidy the Data
    metro_clean <- metro_raw |>
        as_tibble() |>
        rename(metro_current = FULL1_NAME,
               metro_1y = FULL2_NAME,
               moved_in = MOVEDIN,
               moved_out = MOVEDOUT) |>
        mutate(across(c(moved_in, moved_out), as.integer)) |>
        select(metro_current, metro_1y, moved_in, moved_out) |>
        filter(!is.na(metro_current), !is.na(metro_1y))

    return(metro_clean)
}

# 1. DEFINE RENAMED FUNCTION
my_pop_projection_2030 <- function(current_pop, growth, migration_df) {
  natural_growth <- current_pop * (1 + growth)
  
  net_migration <- sapply(names(current_pop), function(s) {
    incoming <- sum(migration_df$m_ij[migration_df$state_current == s] * current_pop[migration_df$state_1y[migration_df$state_current == s]], na.rm = TRUE)
    outgoing <- sum(migration_df$m_ij[migration_df$state_1y == s] * current_pop[s], na.rm = TRUE)
    return(incoming - outgoing)
  })
  
  return(natural_growth + net_migration)
}

fit_model <- function(t0, t1) {
  P_t0 <- sum(state_pops[[as.character(t0)]], na.rm = TRUE)
  P_t1 <- sum(state_pops[[as.character(t1)]], na.rm = TRUE)
  g <- (P_t1 - P_t0) / P_t0
  
  mig_rates <- full_migration_data |>
    filter(year_current == t1, year_1y == t0) |>
    filter(state_current != state_1y) |>
    inner_join(state_pops, by = c("state_current" = "state")) |>
    rename(pop_i = as.character(t0)) |>
    inner_join(state_pops, by = c("state_1y" = "state")) |>
    rename(pop_j = as.character(t0)) |>
    mutate(m_ij = population / sqrt(pop_i * pop_j))
  
  return(list(g = g, m = mig_rates))
}
